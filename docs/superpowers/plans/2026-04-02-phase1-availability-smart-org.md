```typescript
/**
 * Availability data layer — CRUD for athlete availability blocks and overrides,
 * plus "Best Windows" computation for coach team-wide view.
 */

import prisma from "@/lib/prisma";

/* ─── Types ──────────────────────────────────────────────────────────────── */

export type AvailabilityBlock = {
  id: string;
  dayOfWeek: number;
  startTime: string;
  endTime: string;
  type: string;
  label: string | null;
  notes: string | null;
};

export type AvailabilityOverrideItem = {
  id: string;
  date: string;
  startTime: string | null;
  endTime: string | null;
  type: string;
  reason: string | null;
};

export type BestWindow = {
  days: number[];       // [1, 3, 5] for Mon/Wed/Fri
  dayLabels: string[];  // ["Mon", "Wed", "Fri"]
  startTime: string;    // "15:00"
  endTime: string;      // "17:00"
  availableCount: number;
  totalCount: number;
  percentage: number;
  conflictAthletes: { id: string; name: string; reason: string }[];
};

export type AthleteAvailabilitySummary = {
  athleteId: string;
  athleteName: string;
  avatarUrl: string | null;
  hasAvailability: boolean;
  blocks: AvailabilityBlock[];
  overrides: AvailabilityOverrideItem[];
};

/* ─── Constants ──────────────────────────────────────────────────────────── */

const DAY_LABELS = ["Sun", "Mon", "Tue", "Wed", "Thu", "Fri", "Sat"];

const VALID_TYPES = ["AVAILABLE", "UNAVAILABLE", "CONDITIONAL"];
const VALID_OVERRIDE_TYPES = ["AVAILABLE", "UNAVAILABLE"];

/* ─── Athlete CRUD ───────────────────────────────────────────────────────── */

export async function getAthleteAvailability(athleteId: string) {
  const [blocks, overrides] = await Promise.all([
    prisma.athleteAvailability.findMany({
      where: { athleteId },
      orderBy: [{ dayOfWeek: "asc" }, { startTime: "asc" }],
    }),
    prisma.availabilityOverride.findMany({
      where: { athleteId, date: { gte: new Date().toISOString().split("T")[0] } },
      orderBy: { date: "asc" },
    }),
  ]);

  return {
    blocks: blocks.map((b) => ({
      id: b.id,
      dayOfWeek: b.dayOfWeek,
      startTime: b.startTime,
      endTime: b.endTime,
      type: b.type,
      label: b.label,
      notes: b.notes,
    })),
    overrides: overrides.map((o) => ({
      id: o.id,
      date: o.date,
      startTime: o.startTime,
      endTime: o.endTime,
      type: o.type,
      reason: o.reason,
    })),
  };
}

export async function createAvailabilityBlock(
  athleteId: string,
  data: {
    dayOfWeek: number;
    startTime: string;
    endTime: string;
    type: string;
    label?: string | null;
    notes?: string | null;
  },
) {
  if (!VALID_TYPES.includes(data.type)) throw new Error("Invalid type");
  if (data.dayOfWeek < 0 || data.dayOfWeek > 6) throw new Error("Invalid dayOfWeek");

  return prisma.athleteAvailability.create({
    data: {
      athleteId,
      dayOfWeek: data.dayOfWeek,
      startTime: data.startTime,
      endTime: data.endTime,
      type: data.type,
      label: data.label ?? null,
      notes: data.notes ?? null,
    },
  });
}

export async function updateAvailabilityBlock(
  id: string,
  athleteId: string,
  data: Partial<{
    dayOfWeek: number;
    startTime: string;
    endTime: string;
    type: string;
    label: string | null;
    notes: string | null;
  }>,
) {
  if (data.type && !VALID_TYPES.includes(data.type)) throw new Error("Invalid type");

  const block = await prisma.athleteAvailability.findUnique({ where: { id } });
  if (!block || block.athleteId !== athleteId) throw new Error("Not found");

  return prisma.athleteAvailability.update({
    where: { id },
    data: {
      ...(data.dayOfWeek !== undefined && { dayOfWeek: data.dayOfWeek }),
      ...(data.startTime !== undefined && { startTime: data.startTime }),
      ...(data.endTime !== undefined && { endTime: data.endTime }),
      ...(data.type !== undefined && { type: data.type }),
      ...(data.label !== undefined && { label: data.label }),
      ...(data.notes !== undefined && { notes: data.notes }),
    },
  });
}

export async function deleteAvailabilityBlock(id: string, athleteId: string) {
  const block = await prisma.athleteAvailability.findUnique({ where: { id } });
  if (!block || block.athleteId !== athleteId) throw new Error("Not found");
  return prisma.athleteAvailability.delete({ where: { id } });
}

export async function createAvailabilityOverride(
  athleteId: string,
  data: {
    date: string;
    startTime?: string | null;
    endTime?: string | null;
    type: string;
    reason?: string | null;
  },
) {
  if (!VALID_OVERRIDE_TYPES.includes(data.type)) throw new Error("Invalid type");

  return prisma.availabilityOverride.create({
    data: {
      athleteId,
      date: data.date,
      startTime: data.startTime ?? null,
      endTime: data.endTime ?? null,
      type: data.type,
      reason: data.reason ?? null,
    },
  });
}

export async function deleteAvailabilityOverride(id: string, athleteId: string) {
  const override = await prisma.availabilityOverride.findUnique({ where: { id } });
  if (!override || override.athleteId !== athleteId) throw new Error("Not found");
  return prisma.availabilityOverride.delete({ where: { id } });
}

/* ─── Coach: Team Availability Summary ───────────────────────────────────── */

export async function getTeamAvailability(coachId: string, groupId?: string) {
  // Fetch all athletes on roster (optionally filtered by event group)
  const athletes = await prisma.athleteProfile.findMany({
    where: {
      coachId,
      ...(groupId
        ? { eventGroupMemberships: { some: { groupId } } }
        : {}),
    },
    select: {
      id: true,
      firstName: true,
      lastName: true,
      avatarUrl: true,
      availability: {
        orderBy: [{ dayOfWeek: "asc" }, { startTime: "asc" }],
      },
      availabilityOverrides: {
        where: { date: { gte: new Date().toISOString().split("T")[0] } },
        orderBy: { date: "asc" },
      },
    },
    orderBy: [{ lastName: "asc" }, { firstName: "asc" }],
  });

  const summaries: AthleteAvailabilitySummary[] = athletes.map((a) => ({
    athleteId: a.id,
    athleteName: `${a.firstName} ${a.lastName}`,
    avatarUrl: a.avatarUrl,
    hasAvailability: a.availability.length > 0,
    blocks: a.availability.map((b) => ({
      id: b.id,
      dayOfWeek: b.dayOfWeek,
      startTime: b.startTime,
      endTime: b.endTime,
      type: b.type,
      label: b.label,
      notes: b.notes,
    })),
    overrides: a.availabilityOverrides.map((o) => ({
      id: o.id,
      date: o.date,
      startTime: o.startTime,
      endTime: o.endTime,
      type: o.type,
      reason: o.reason,
    })),
  }));

  // Compute best windows
  const bestWindows = computeBestWindows(summaries);

  return { athletes: summaries, bestWindows, totalAthletes: athletes.length };
}

/* ─── Best Windows Computation ───────────────────────────────────────────── */

/**
 * Computes the best practice time windows by intersecting all athletes'
 * AVAILABLE blocks. Groups consecutive 30-min slots with the same
 * available count into windows. Returns top 10 windows with ≥50% availability.
 */
function computeBestWindows(athletes: AthleteAvailabilitySummary[]): BestWindow[] {
  if (athletes.length === 0) return [];

  const total = athletes.length;

  // Build a map: dayOfWeek → slot (e.g., "14:00") → list of available athlete IDs
  const slotMap = new Map<string, Set<string>>(); // key: "day:HH:MM"
  const conflictMap = new Map<string, { id: string; name: string; reason: string }[]>();

  // Generate 30-min slots from 06:00 to 22:00 for each day
  for (let day = 0; day <= 6; day++) {
    for (let hour = 6; hour < 22; hour++) {
      for (const min of [0, 30]) {
        const slot = `${day}:${String(hour).padStart(2, "0")}:${String(min).padStart(2, "0")}`;
        slotMap.set(slot, new Set());
        conflictMap.set(slot, []);
      }
    }
  }

  // For each athlete, mark which slots they're available for
  for (const athlete of athletes) {
    const availableSlots = new Set<string>();
    const unavailableSlots = new Map<string, string>(); // slot → reason

    for (const block of athlete.blocks) {
      const slots = expandBlockToSlots(block.dayOfWeek, block.startTime, block.endTime);
      for (const slot of slots) {
        if (block.type === "AVAILABLE") {
          availableSlots.add(slot);
        } else if (block.type === "UNAVAILABLE" || block.type === "CONDITIONAL") {
          unavailableSlots.set(slot, block.label || block.type);
          availableSlots.delete(slot);
        }
      }
    }

    for (const slot of availableSlots) {
      slotMap.get(slot)?.add(athlete.athleteId);
    }

    // Track conflicts (slots where athlete is NOT available)
    for (const [slot, reason] of unavailableSlots) {
      conflictMap.get(slot)?.push({
        id: athlete.athleteId,
        name: athlete.athleteName,
        reason,
      });
    }
  }

  // Group by day pattern: find day combinations that share similar availability
  // Simpler approach: rank individual day+time slots, then merge consecutive ones
  const daySlots = new Map<number, { time: string; count: number }[]>();

  for (const [key, athleteIds] of slotMap) {
    const [dayStr, ...timeParts] = key.split(":");
    const day = parseInt(dayStr);
    const time = timeParts.join(":");
    const count = athleteIds.size;

    if (count === 0) continue;

    if (!daySlots.has(day)) daySlots.set(day, []);
    daySlots.get(day)!.push({ time, count });
  }

  // Build windows: consecutive slots on same day with ≥50% availability
  const windows: BestWindow[] = [];
  const threshold = Math.ceil(total * 0.5);

  for (const [day, slots] of daySlots) {
    const sorted = slots.filter((s) => s.count >= threshold).sort((a, b) => a.time.localeCompare(b.time));
    if (sorted.length === 0) continue;

    let windowStart = sorted[0].time;
    let windowMinCount = sorted[0].count;
    let prevTime = sorted[0].time;

    for (let i = 1; i <= sorted.length; i++) {
      const current = sorted[i];
      const prevMinutes = timeToMinutes(prevTime);
      const currentMinutes = current ? timeToMinutes(current.time) : -1;

      // If gap or end of array, close window
      if (!current || currentMinutes - prevMinutes > 30) {
        const endMinutes = timeToMinutes(prevTime) + 30;
        const endTime = minutesToTime(endMinutes);
        const slotKey = `${day}:${windowStart}`;
        const conflicts = conflictMap.get(slotKey) || [];

        windows.push({
          days: [day],
          dayLabels: [DAY_LABELS[day]],
          startTime: windowStart,
          endTime,
          availableCount: windowMinCount,
          totalCount: total,
          percentage: Math.round((windowMinCount / total) * 100),
          conflictAthletes: conflicts,
        });

        if (current) {
          windowStart = current.time;
          windowMinCount = current.count;
        }
      } else {
        windowMinCount = Math.min(windowMinCount, current.count);
      }

      if (current) prevTime = current.time;
    }
  }

  // Sort by percentage descending, then by window duration descending
  windows.sort((a, b) => {
    if (b.percentage !== a.percentage) return b.percentage - a.percentage;
    const aDuration = timeToMinutes(a.endTime) - timeToMinutes(a.startTime);
    const bDuration = timeToMinutes(b.endTime) - timeToMinutes(b.startTime);
    return bDuration - aDuration;
  });

  return windows.slice(0, 10);
}

/* ─── Helpers ────────────────────────────────────────────────────────────── */

function expandBlockToSlots(dayOfWeek: number, startTime: string, endTime: string): string[] {
  const slots: string[] = [];
  let current = timeToMinutes(startTime);
  const end = timeToMinutes(endTime);

  while (current < end) {
    const time = minutesToTime(current);
    slots.push(`${dayOfWeek}:${time}`);
    current += 30;
  }

  return slots;
}

function timeToMinutes(time: string): number {
  const [h, m] = time.split(":").map(Number);
  return h * 60 + m;
}

function minutesToTime(minutes: number): string {
  const h = Math.floor(minutes / 60);
  const m = minutes % 60;
  return `${String(h).padStart(2, "0")}:${String(m).padStart(2, "0")}`;
}
```

```typescript
import { NextRequest, NextResponse } from "next/server";
import { getSession } from "@/lib/auth";
import prisma from "@/lib/prisma";
import { logger } from "@/lib/logger";
import {
  getAthleteAvailability,
  createAvailabilityBlock,
} from "@/lib/data/availability";

export async function GET() {
  try {
    const session = await getSession();
    if (!session) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

    const athlete = await prisma.athleteProfile.findUnique({
      where: { userId: session.userId },
      select: { id: true },
    });
    if (!athlete) return NextResponse.json({ error: "Athlete not found" }, { status: 404 });

    const data = await getAthleteAvailability(athlete.id);
    return NextResponse.json({ success: true, data });
  } catch (err) {
    logger.error("GET athlete availability error", { error: err });
    return NextResponse.json({ error: "Failed to fetch availability" }, { status: 500 });
  }
}

export async function POST(req: NextRequest) {
  try {
    const session = await getSession();
    if (!session) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

    const athlete = await prisma.athleteProfile.findUnique({
      where: { userId: session.userId },
      select: { id: true },
    });
    if (!athlete) return NextResponse.json({ error: "Athlete not found" }, { status: 404 });

    const body = await req.json();
    const { dayOfWeek, startTime, endTime, type, label, notes } = body;

    if (typeof dayOfWeek !== "number" || !startTime || !endTime || !type) {
      return NextResponse.json({ error: "dayOfWeek, startTime, endTime, and type are required" }, { status: 400 });
    }

    const block = await createAvailabilityBlock(athlete.id, {
      dayOfWeek,
      startTime,
      endTime,
      type,
      label,
      notes,
    });

    return NextResponse.json({ success: true, data: block });
  } catch (err) {
    logger.error("POST athlete availability error", { error: err });
    const message = err instanceof Error ? err.message : "Failed to create block";
    return NextResponse.json({ error: message }, { status: 500 });
  }
}
```

```typescript
import { NextRequest, NextResponse } from "next/server";
import { requireCoachApi, AuthError } from "@/lib/data/coach";

export async function GET(req: NextRequest) {
```

Create `src/app/api/coach/availability/[athleteId]/route.ts`:

```typescript
import { NextRequest, NextResponse } from "next/server";
import { requireCoachApi, AuthError } from "@/lib/data/coach";
import { canAccessAthlete } from "@/lib/authorize";
import { logger } from "@/lib/logger";
import { getAthleteAvailability } from "@/lib/data/availability";

export async function GET(
  _req: NextRequest,
  { params }: { params: Promise<{ athleteId: string }> },
) {
  try {
    const { session, coach } = await requireCoachApi();
    const { athleteId } = await params;

    if (!(await canAccessAthlete(session.userId, "COACH", athleteId))) {
      return NextResponse.json({ error: "Forbidden" }, { status: 403 });
    }

    const data = await getAthleteAvailability(athleteId);
    return NextResponse.json({ success: true, data });
  } catch (err) {
    if (err instanceof AuthError) {
      return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
    }
    logger.error("GET coach single athlete availability error", { error: err });
    return NextResponse.json({ error: "Failed to fetch availability" }, { status: 500 });
  }
}
```
